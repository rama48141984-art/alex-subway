<script>
const canvas = document.getElementById('gameCanvas');
const ctx = canvas.getContext('2d');
const levelDisplay = document.getElementById('levelDisplay');
const victoryScreen = document.getElementById('victoryScreen');

function resizeCanvas() {
    canvas.width = canvas.parentElement.clientWidth;
    canvas.height = canvas.parentElement.clientHeight;
}
resizeCanvas();
window.addEventListener('resize', resizeCanvas);

const gravity = 0.4;
const friction = 0.85;

const player = {
    x: 100, y: 0,
    width: 30, height: 45,
    vx: 0, vy: 0,
    speed: 5.5, jumpForce: 11.5,
    grounded: false, squishX: 1, squishY: 1, angle: 0
};

const keys = { left: false, right: false, up: false };
const camera = { y: 0 };

// NEW PLATFORM TYPES
const PLATFORM_TYPES = ['mushroom', 'candy', 'gear', 'carpet'];

let platforms = [];
const TOTAL_LEVELS = 7;

function generateLevelData() {
    platforms = [];
    
    // Safety Base Ground
    platforms.push({ x: 0, y: canvas.height - 40, width: 2000, height: 100, type: 'ground' });

    let currentY = canvas.height - 140;
    
    for (let lvl = 1; lvl <= TOTAL_LEVELS; lvl++) {
        let currentX = 150 + Math.random() * 100;
        
        for (let i = 0; i < 6; i++) {
            let type = PLATFORM_TYPES[Math.floor(Math.random() * PLATFORM_TYPES.length)];
            let pWidth = 80;
            let pHeight = 25;
            
            // Set dimensions based on the new types
            if (type === 'mushroom') { pWidth = 60; pHeight = 25; }
            if (type === 'candy') { pWidth = 85; pHeight = 20; }
            if (type === 'gear') { pWidth = 60; pHeight = 25; }
            if (type === 'carpet') { pWidth = 90; pHeight = 10; }

            platforms.push({
                x: currentX,
                y: currentY,
                width: pWidth,
                height: pHeight,
                type: type,
                level: lvl,
                // Mushrooms bounce you higher, carpets are slightly softer
                bounce: type === 'mushroom' ? 1.4 : (type === 'carpet' ? 0.9 : 1.0),
                animOffset: Math.random() * Math.PI * 2
            });

            let dir = Math.random() > 0.5 ? 1 : -1;
            currentX += dir * (130 + Math.random() * 110);
            
            if (currentX < 40) currentX = 80 + Math.random() * 50;
            if (currentX > canvas.width - 120) currentX = canvas.width - 200 - Math.random() * 50;
            
            currentY -= 115; 
        }
    }

    platforms.push({
        x: canvas.width / 2 - 50, y: currentY - 50,
        width: 90, height: 50, type: 'crown', level: 7
    });
}

window.addEventListener('keydown', (e) => {
    if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') keys.left = true;
    if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') keys.right = true;
    if (e.key === 'ArrowUp' || e.key === 'w' || e.key === 'W' || e.key === ' ') keys.up = true;
});
window.addEventListener('keyup', (e) => {
    if (e.key === 'ArrowLeft' || e.key === 'a' || e.key === 'A') keys.left = false;
    if (e.key === 'ArrowRight' || e.key === 'd' || e.key === 'D') keys.right = false;
    if (e.key === 'ArrowUp' || e.key === 'w' || e.key === 'W' || e.key === ' ') keys.up = false;
});

function setupMobileControls() {
    const bindBtn = (id, action) => {
        const btn = document.getElementById(id);
        btn.addEventListener('touchstart', (e) => { e.preventDefault(); keys[action] = true; }, {passive: false});
        btn.addEventListener('touchend', (e) => { e.preventDefault(); keys[action] = false; }, {passive: false});
    };
    bindBtn('btnLeft', 'left');
    bindBtn('btnRight', 'right');
    bindBtn('btnJump', 'up');
}
setupMobileControls();

function resetGame() {
    victoryScreen.style.display = 'none';
    player.x = 100; player.y = canvas.height - 150;
    player.vx = 0; player.vy = 0;
    camera.y = 0;
    generateLevelData();
}

let particles = [];
for(let i=0; i<60; i++) {
    particles.push({
        x: Math.random() * 800, y: Math.random() * 3000,
        size: Math.random() * 2 + 1, speed: Math.random() * 0.5 + 0.2
    });
}

let animationTime = 0;

function update() {
    animationTime += 0.04;

    if (keys.left) player.vx = -player.speed;
    else if (keys.right) player.vx = player.speed;
    else player.vx *= friction;

    player.vy += gravity;
    player.x += player.vx;
    player.y += player.vy;

    if (player.x < 0) player.x = 0;
    if (player.x + player.width > canvas.width) player.x = canvas.width - player.width;

    if (player.y > canvas.height) {
        player.x = 100; player.y = canvas.height - 150;
        player.vx = 0; player.vy = 0;
    }

    let targetCameraY = player.y - canvas.height * 0.6;
    camera.y += (targetCameraY - camera.y) * 0.1; 

    if (!player.grounded) {
        player.squishX += (0.9 - player.squishX) * 0.1;
        player.squishY += (1.1 - player.squishY) * 0.1;
        if(player.vx !== 0) player.angle += (player.vx > 0 ? 0.05 : -0.05);
    } else {
        player.squishX += (1.0 - player.squishX) * 0.2;
        player.squishY += (1.0 - player.squishY) * 0.2;
        player.angle *= 0.8;
    }

    player.grounded = false;
    let highestLevelReached = 1;

    platforms.forEach(p => {
        if (player.y < p.y && p.level > highestLevelReached) {
            highestLevelReached = p.level;
        }

        if (player.x + player.width > p.x &&
            player.x < p.x + p.width &&
            player.y + player.height > p.y &&
            player.y + player.height - player.vy <= p.y + 12) {
            
            if (player.vy > 0) {
                if (p.type === 'crown') {
                    victoryScreen.style.display = 'flex';
                    return;
                }
                
                player.grounded = true;
                player.y = p.y - player.height;
                
                if (keys.up) {
                    player.vy = -player.jumpForce * (p.bounce || 1.0);
                    player.squishX = 0.7; player.squishY = 1.4;
                } else {
                    player.vy = 0;
                }
            }
        }
    });

    levelDisplay.textContent = Math.min(highestLevelReached, TOTAL_LEVELS);

    if (keys.up && player.grounded) {
        player.vy = -player.jumpForce;
        player.grounded = false;
    }

    render();
    requestAnimationFrame(update);
}

function render() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    ctx.save();
    ctx.translate(0, -camera.y);

    ctx.fillStyle = "rgba(0, 255, 255, 0.4)";
    particles.forEach(p => {
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fill();
    });

    platforms.forEach(p => {
        ctx.save();
        let dynamicSway = Math.sin(animationTime + p.animOffset) * 4;

        if (p.type === 'ground') {
            ctx.fillStyle = '#0a2342';
            ctx.fillRect(p.x, p.y, p.width, p.height);
            ctx.strokeStyle = '#00ffff';
            ctx.lineWidth = 4;
            ctx.strokeRect(p.x, p.y, p.width, p.height);
        } 
        
        // NEW OBSTACLE: MUSHROOM
        else if (p.type === 'mushroom') {
            ctx.translate(p.x + p.width/2, p.y + dynamicSway);
            // Stalk
            ctx.fillStyle = '#f0e6d2';
            ctx.fillRect(-10, 0, 20, p.height);
            // Red Cap
            ctx.fillStyle = '#ff3333';
            ctx.beginPath();
            ctx.arc(0, 0, p.width/2 + 5, Math.PI, Math.PI * 2);
            ctx.fill();
            // White Dots
            ctx.fillStyle = '#ffffff';
            ctx.beginPath(); ctx.arc(-12, -8, 5, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(14, -6, 4, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(0, -14, 6, 0, Math.PI*2); ctx.fill();
        } 
        
        // NEW OBSTACLE: CANDY
        else if (p.type === 'candy') {
            ctx.translate(p.x, p.y + dynamicSway * 0.5);
            ctx.fillStyle = '#ff99cc';
            ctx.beginPath();
            ctx.roundRect(0, 0, p.width, p.height, 10);
            ctx.fill();
            // White Stripes
            ctx.strokeStyle = '#ffffff';
            ctx.lineWidth = 5;
            for(let s = 15; s < p.width; s += 20) {
                ctx.beginPath(); ctx.moveTo(s, 0); ctx.lineTo(s - 10, p.height); ctx.stroke();
            }
        } 
        
        // NEW OBSTACLE: ROTATING GEAR
        else if (p.type === 'gear') {
            ctx.translate(p.x + p.width/2, p.y + p.height/2);
            ctx.rotate(animationTime + p.animOffset); // Continuous rotation
            ctx.fillStyle = '#8899aa';
            // Gear teeth
            ctx.beginPath();
            for(let i=0; i<8; i++) {
                ctx.lineTo(0, -p.width/2);
                ctx.rotate(Math.PI/8);
                ctx.lineTo(0, -p.width/2 + 8);
                ctx.rotate(Math.PI/8);
            }
            ctx.closePath();
            ctx.fill();
            // Center hole
            ctx.fillStyle = '#223344';
            ctx.beginPath(); ctx.arc(0, 0, 8, 0, Math.PI*2); ctx.fill();
        } 
        
        // NEW OBSTACLE: MAGIC CARPET
        else if (p.type === 'carpet') {
            // Wavy up and down motion
            ctx.translate(p.x, p.y + Math.sin(animationTime * 2 + p.animOffset) * 12); 
            ctx.fillStyle = '#8a2be2'; // Purple
            ctx.fillRect(0, 0, p.width, p.height);
            // Gold Trim
            ctx.fillStyle = '#ffd700'; 
            ctx.fillRect(0, 0, p.width, 3);
            ctx.fillRect(0, p.height - 3, p.width, 3);
            // Tassels
            ctx.beginPath(); ctx.moveTo(0, 0); ctx.lineTo(-8, 5); ctx.lineTo(0, p.height); ctx.fill();
            ctx.beginPath(); ctx.moveTo(p.width, 0); ctx.lineTo(p.width + 8, 5); ctx.lineTo(p.width, p.height); ctx.fill();
        }
        
        else if (p.type === 'crown') {
            ctx.translate(p.x, p.y + Math.sin(animationTime * 2) * 5);
            ctx.fillStyle = '#ffcc00';
            ctx.beginPath();
            ctx.moveTo(0, p.height);
            ctx.lineTo(0, 10);
            ctx.lineTo(p.width * 0.25, p.height * 0.4);
            ctx.lineTo(p.width * 0.5, 0);
            ctx.lineTo(p.width * 0.75, p.height * 0.4);
            ctx.lineTo(p.width, 10);
            ctx.lineTo(p.width, p.height);
            ctx.closePath();
            ctx.fill();
            ctx.fillStyle = '#00ffff';
            ctx.beginPath(); ctx.arc(p.width*0.5, 5, 5, 0, Math.PI*2); ctx.fill();
        }
        
        ctx.restore();
    });

    // Draw Player
    ctx.save();
    ctx.translate(player.x + player.width / 2, player.y + player.height / 2);
    ctx.rotate(player.angle);
    ctx.scale(player.squishX, player.squishY);

    ctx.fillStyle = '#00ffff';
    ctx.beginPath();
    ctx.roundRect(-player.width / 2, -player.height / 2, player.width, player.height, 8);
    ctx.fill();

    ctx.fillStyle = '#ffffff';
    let eyeDir = keys.right ? 3 : (keys.left ? -3 : 0);
    ctx.beginPath();
    ctx.arc(-6 + eyeDir, -8, 5, 0, Math.PI * 2);
    ctx.arc(6 + eyeDir, -8, 5, 0, Math.PI * 2);
    ctx.fill();
    ctx.fillStyle = '#000000';
    ctx.beginPath();
    ctx.arc(-6 + eyeDir * 1.3, -8, 2.5, 0, Math.PI * 2);
    ctx.arc(6 + eyeDir * 1.3, -8, 2.5, 0, Math.PI * 2);
    ctx.fill();

    ctx.fillStyle = 'rgba(0, 100, 255, 0.5)';
    ctx.beginPath(); ctx.arc(-10 + eyeDir, -2, 3, 0, Math.PI * 2); ctx.fill();
    ctx.beginPath(); ctx.arc(10 + eyeDir, -2, 3, 0, Math.PI * 2); ctx.fill();

    ctx.restore();
    ctx.restore();
}

resetGame();
update();
</script>
